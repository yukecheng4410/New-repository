# New-repository
- /src
    - /frontend
        - /components  # 存放 Vue/React 组件
        - /pages  # 页面组件，路由对应的页面
        - /utils  # 工具函数目录
        - main.js  # 前端入口文件
    - /backend
        - /controllers  # 业务逻辑控制层，处理 API 请求
        - /models  # 数据模型定义，对应数据库表结构
        - /routes  # 路由定义，关联 URL 与控制器方法
        - app.py  # 后端入口文件（以 Python Flask 为例）
- /tests
    - /frontend  # 前端测试用例目录
    - /backend  # 后端测试用例目录
- README.md
- requirements.txt  # 后端依赖文件
- package.json  # 前端依赖文件（前端为 JavaScript 项目时）
